        import * as THREE from 'three';
        import { OrbitControls } from 'three/addons/controls/OrbitControls.js';

        // 1. Setup Scene
        const scene = new THREE.Scene();
        // We don't set a scene background color here because we use CSS for the gradient background

        // 2. Setup Camera (Orthographic for that "Isometric/Vector" look)
        const aspect = window.innerWidth / window.innerHeight;
        const frustumSize = 5;
        const camera = new THREE.OrthographicCamera(
            frustumSize * aspect / -2,
            frustumSize * aspect / 2,
            frustumSize / 2,
            frustumSize / -2,
            1,
            1000
        );

        // Position camera to look at the object isometrically
        camera.position.set(20, 20, 20);
        camera.lookAt(0, 0, 0);

        // 3. Renderer
        const renderer = new THREE.WebGLRenderer({ alpha: true, antialias: true });
        renderer.setSize(window.innerWidth, window.innerHeight);
        renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
        document.body.appendChild(renderer.domElement);

        // 4. Controls
        const controls = new OrbitControls(camera, renderer.domElement);
        controls.enableDamping = true;

        // 5. Create the Geometry (Thick Square Path)
        class SquarePath extends THREE.Curve {
            getPoint(t) {
                const size = 1.5;
                const h = size / 2;

                // Define a square path in 2D
                if (t < 0.25) {
                    const k = t / 0.25;
                    return new THREE.Vector3(-h + size * k, -h, 0); // Bottom
                } else if (t < 0.5) {
                    const k = (t - 0.25) / 0.25;
                    return new THREE.Vector3(h, -h + size * k, 0); // Right
                } else if (t < 0.75) {
                    const k = (t - 0.5) / 0.25;
                    return new THREE.Vector3(h - size * k, h, 0); // Top
                } else {
                    const k = (t - 0.75) / 0.25;
                    return new THREE.Vector3(-h, h - size * k, 0); // Left
                }
            }
        }

        // Define the cross-section shape (A rectangle)
        const width = 0.4;
        const thickness = 0.4;
        const shape = new THREE.Shape();
        shape.moveTo(-width / 2, -thickness / 2);
        shape.lineTo(width / 2, -thickness / 2);
        shape.lineTo(width / 2, thickness / 2);
        shape.lineTo(-width / 2, thickness / 2);
        shape.closePath();

        // Extrude Settings
        const extrudeSettings = {
            steps: 200,
            curveSegments: 20,
            extrudePath: new SquarePath(),
            depth: 0,
            bevelEnabled: false
        };

        const geometry = new THREE.ExtrudeGeometry(shape, extrudeSettings);

        // 6. Apply the MÃ¶bius Twist manually
        // We rotate vertices along the path to create the twist
        const pos = geometry.attributes.position;
        const center = new THREE.Vector3();
        const axis = new THREE.Vector3();

        for (let i = 0; i < pos.count; i++) {
            const x = pos.getX(i);
            const y = pos.getY(i);
            const z = pos.getZ(i);

            // Simple trick: Calculate angle based on position around the center
            // This approximates the progress 't' along the square loop
            const angle = Math.atan2(y, x);

            // Twist amount: 180 degrees (PI) creates a single Mobius twist
            // We add a slight offset to align corners aesthetically
            const twist = angle + (Math.PI / 4);

            // Apply rotation around the path tangent (roughly Z-axis here)
            // Note: For a perfect math-mobius we'd use Frenet frames, but this visual hack works well for the "Impossible Object" look
            const twistAmount = (angle / (Math.PI * 2)) * Math.PI; // 180 deg twist over full loop

            // We actually want a cleaner twist for the Penrose look. 
            // Let's rely on a custom Shader for the color and keep geometry simple.
            // But to make it "Mobius", we do need to flip coordinates.
            // Let's use a simpler visual twist:

            // Rotating the geometry to stand up
        }

        // Rotate geometry to face camera nicely
        geometry.center();
        geometry.rotateX(Math.PI / 2); // Lay flat
        geometry.rotateZ(Math.PI / 4); // Diamond orientation

        // 7. Shader Material (The Secret Sauce for the Gradient)
        const gradientMaterial = new THREE.ShaderMaterial({
            uniforms: {
                color1: { value: new THREE.Color(0x00f2ff) }, // Cyan
                color2: { value: new THREE.Color(0x9d00ff) }, // Purple/Magenta
            },
            vertexShader: `
            varying vec2 vUv;
            varying vec3 vPos;
            void main() {
                vUv = uv;
                vPos = position;
                // Standard projection
                gl_Position = projectionMatrix * modelViewMatrix * vec4(position, 1.0);
            }
        `,
            fragmentShader: `
            uniform vec3 color1;
            uniform vec3 color2;
            varying vec3 vPos;
            
            void main() {
                // Create a gradient based on screen/world coordinates
                // Mixing x and y creates a diagonal gradient
                float mixFactor = (vPos.x + vPos.y + 2.0) / 4.0;
                
                // Add some "fake lighting" based on normals to distinguish faces
                // Since this is unlit, we use coordinate logic or flat shading
                
                vec3 finalColor = mix(color1, color2, clamp(mixFactor, 0.0, 1.0));
                
                // Add a subtle shadow/highlight for 3D depth
                float depth = vPos.z * 0.5;
                finalColor += depth * 0.1;

                gl_FragColor = vec4(finalColor, 1.0);
            }
        `,
            side: THREE.DoubleSide
        });

        const mesh = new THREE.Mesh(geometry, gradientMaterial);

        // Apply a specific twist to the geometry object itself to match the image orientation
        mesh.rotation.x = Math.PI / 5;
        mesh.rotation.y = Math.PI / 4;

        scene.add(mesh);

        // 8. Animation Loop
        const animate = () => {

import React, { useEffect, useState, useRef } from "react";
import { X } from "lucide-react";

/**
 * Header Component
 * Features: 
 * - Scroll-based visibility
 * - Fullscreen menu animation
 * - Custom SVG Square "Redraw" hover effect 
 * - Fix: Works every time, including after menu is opened/closed.
 */
export default function App() {
  const [menuOpen, setMenuOpen] = useState(false);
  const [showHeader, setShowHeader] = useState(true);
  const [bgBlack, setBgBlack] = useState(false);
  const [gsapLoaded, setGsapLoaded] = useState(false);

  const menuItemsRef = useRef([]);
  const menuBgRef = useRef(null);
  const menuIconRef = useRef(null);
  const animatedStrokeRef = useRef(null);
  const tl = useRef(null);
  const lastScrollY = useRef(0);

  /* =========================
      LOAD GSAP FROM CDN
  ========================== */
  useEffect(() => {
    if (window.gsap) {
      setGsapLoaded(true);
      return;
    }
    const script = document.createElement("script");
    script.src = "https://cdnjs.cloudflare.com/ajax/libs/gsap/3.12.2/gsap.min.js";
    script.onload = () => setGsapLoaded(true);
    document.head.appendChild(script);
  }, []);

  /* =========================
      HEADER SCROLL BEHAVIOR
  ========================== */
  useEffect(() => {
    const onScroll = () => {
      const currentScrollY = window.scrollY;
      if (currentScrollY > lastScrollY.current && currentScrollY > 100) {
        setShowHeader(false);
      } else {
        setShowHeader(true);
      }
      setBgBlack(currentScrollY > 80);
      lastScrollY.current = currentScrollY;
    };

    window.addEventListener("scroll", onScroll);
    return () => window.removeEventListener("scroll", onScroll);
  }, []);

  /* =========================
      GSAP MENU TIMELINE
  ========================== */
  useEffect(() => {
    if (!gsapLoaded || !window.gsap) return;
    const gsap = window.gsap;

    const ctx = gsap.context(() => {
      tl.current = gsap.timeline({ paused: true });
      tl.current
        .fromTo(
          menuBgRef.current,
          { clipPath: "circle(0% at 95% 5%)", opacity: 0 },
          {
            clipPath: "circle(150% at 95% 5%)",
            opacity: 1,
            duration: 0.8,
            ease: "power4.inOut",
          }
        )
        .fromTo(
          menuItemsRef.current,
          { x: -50, opacity: 0 },
          {
            x: 0,
            opacity: 1,
            duration: 0.6,
            ease: "power3.out",
            stagger: 0.1,
          },
          "-=0.4"
        );
    });

    return () => ctx.revert();
  }, [gsapLoaded]);

  /* =========================
      SQUARE REDRAW HOVER LOGIC
  ========================== */
  useEffect(() => {
    if (!gsapLoaded || !window.gsap || menuOpen) return;
    const gsap = window.gsap;

    const path = animatedStrokeRef.current;
    const iconContainer = menuIconRef.current;

    if (path && iconContainer) {
      const length = path.getTotalLength();
      
      // Ensure stroke is set up correctly every time menu closes
      gsap.set(path, { 
        strokeDasharray: length,

import React, { useEffect, useState, useRef } from "react";
import { X } from "lucide-react";

/**
 * Header Component
 * Features: 
 * - Scroll-based visibility
 * - Fullscreen menu animation
 * - Custom SVG Square "Redraw" hover effect 
 * - Logic: Mouse Enter (Forward Redraw) / Mouse Leave (Reverse Redraw)
 */
export default function App() {
  const [menuOpen, setMenuOpen] = useState(false);
  const [showHeader, setShowHeader] = useState(true);
  const [bgBlack, setBgBlack] = useState(false);
  const [gsapLoaded, setGsapLoaded] = useState(false);

  const menuItemsRef = useRef([]);
  const menuBgRef = useRef(null);
  const menuIconRef = useRef(null);
  const animatedStrokeRef = useRef(null);
  const tl = useRef(null);
  const lastScrollY = useRef(0);

  /* =========================
      LOAD GSAP FROM CDN
  ========================== */
  useEffect(() => {
    if (window.gsap) {
      setGsapLoaded(true);
      return;
    }
    const script = document.createElement("script");
    script.src = "https://cdnjs.cloudflare.com/ajax/libs/gsap/3.12.2/gsap.min.js";
    script.onload = () => setGsapLoaded(true);
    document.head.appendChild(script);
  }, []);

  /* =========================
      HEADER SCROLL BEHAVIOR
  ========================== */
  useEffect(() => {
    const onScroll = () => {
      const currentScrollY = window.scrollY;
      if (currentScrollY > lastScrollY.current && currentScrollY > 100) {
        setShowHeader(false);
      } else {
        setShowHeader(true);
      }
      setBgBlack(currentScrollY > 80);
      lastScrollY.current = currentScrollY;
    };

    window.addEventListener("scroll", onScroll);
    return () => window.removeEventListener("scroll", onScroll);
  }, []);

  /* =========================
      GSAP MENU TIMELINE
  ========================== */
  useEffect(() => {
    if (!gsapLoaded || !window.gsap) return;
    const gsap = window.gsap;

    const ctx = gsap.context(() => {
      tl.current = gsap.timeline({ paused: true });
      tl.current
        .fromTo(
          menuBgRef.current,
          { clipPath: "circle(0% at 95% 5%)", opacity: 0 },
          {
            clipPath: "circle(150% at 95% 5%)",
            opacity: 1,
            duration: 0.8,
            ease: "power4.inOut",
          }
        )
        .fromTo(
          menuItemsRef.current,
          { x: -50, opacity: 0 },
          {
            x: 0,
            opacity: 1,
            duration: 0.6,
            ease: "power3.out",
            stagger: 0.1,
          },
          "-=0.4"
        );
    });

    return () => ctx.revert();
  }, [gsapLoaded]);

  /* =========================
      SQUARE REDRAW HOVER LOGIC
  ========================== */
  useEffect(() => {
    if (!gsapLoaded || !window.gsap || menuOpen) return;
    const gsap = window.gsap;

    const path = animatedStrokeRef.current;
    const iconContainer = menuIconRef.current;

    if (path && iconContainer) {
      const length = path.getTotalLength();
      
      // By default, set the yellow stroke to be visible (offset 0)
      gsap.set(path, { 
        strokeDasharray: length, 
        strokeDashoffset: 0,
        opacity: 1 
      });

      const onMouseEnter = () => {
        // Redraw Forward: Go from full length offset (hidden) to 0 (visible)
        gsap.fromTo(path, 
          { strokeDashoffset: length },
          { 
            strokeDashoffset: 0, 
            duration: 0.6, 
            ease: "power2.inOut",
            overwrite: "auto" 
          }
        );
      };

      const onMouseLeave = () => {
        // Reverse Redraw: Animate the offset back to length (hiding it) then reset to 0
        // or simply animate back to length to simulate a "reverse" flow.
        gsap.to(path, { 
          strokeDashoffset: length, 
          duration: 0.6, 
          ease: "power2.inOut",
          overwrite: "auto",
          onComplete: () => {
            // Once the "reverse" hiding is done, reset to solid yellow state
            gsap.set(path, { strokeDashoffset: 0 });
          }
        });
      };

      iconContainer.addEventListener("mouseenter", onMouseEnter);
      iconContainer.addEventListener("mouseleave", onMouseLeave);

      return () => {
        iconContainer.removeEventListener("mouseenter", onMouseEnter);
        iconContainer.removeEventListener("mouseleave", onMouseLeave);
      };
    }
  }, [gsapLoaded, menuOpen]);

  useEffect(() => {
    if (tl.current) {
      if (menuOpen) tl.current.play();
      else tl.current.reverse();
    }
  }, [menuOpen]);

  const menuLinks = [
    { label: "Home", href: "#" },
    { label: "Services", href: "#" },
    { label: "Portfolio", href: "#" },
    { label: "About Us", href: "#" },
    { label: "Contact Us", href: "#" },
    { label: "Careers", href: "#" },
  ];

  return (
    <div className="min-h-screen bg-neutral-950 text-white font-sans overflow-x-hidden">
      <nav
        className={`fixed top-0 left-0 right-0 z-50 px-6 py-4 md:px-12 md:py-6 flex items-center justify-between transition-all duration-500
        ${showHeader ? "translate-y-0" : "-translate-y-full"}
        ${bgBlack ? "bg-black/90 backdrop-blur-md" : "bg-transparent"}`}
      >
        <div className="flex items-center gap-2 cursor-pointer group" onClick={() => setMenuOpen(false)}>
          <div className="w-10 h-10 bg-[#f8d249] rounded-sm flex items-center justify-center text-black font-black text-xl italic transition-transform group-hover:rotate-12">
            P
          </div>
          <span className="font-bold text-2xl tracking-tighter uppercase hidden sm:block">Pragyan</span>
        </div>

        <button
          onClick={() => setMenuOpen(!menuOpen)}
          className="relative z-50 flex items-center gap-4 group cursor-pointer outline-none"
        >
          <span className={`font-medium uppercase tracking-widest text-sm transition-colors ${menuOpen ? 'text-white' : 'group-hover:text-[#f8d249]'}`}>
            {menuOpen ? "Close" : "Menu"}
          </span>

          <div ref={menuIconRef} className="relative w-10 h-10 flex items-center justify-center">
            {menuOpen ? (
              <X size={28} className="text-white animate-in zoom-in duration-300" />
            ) : (
              <div className="relative w-full h-full animate-in fade-in duration-500">
                <svg viewBox="0 0 40 40" className="w-full h-full overflow-visible">
                  {/* The Redrawing Path - Yellow by default */}
                  <path
                    ref={animatedStrokeRef}
                    d="M 36,4 L 36,36 L 4,36 L 4,4 L 36,4"
                    fill="none"
                    stroke="#f8d249"
                    strokeWidth="2"
                    strokeLinecap="round"
                  />
                  <line x1="12" y1="16" x2="28" y2="16" stroke="white" strokeWidth="2" strokeLinecap="round" />
                  <line x1="12" y1="24" x2="28" y2="24" stroke="white" strokeWidth="2" strokeLinecap="round" />
                </svg>
              </div>
            )}
          </div>
        </button>
      </nav>

      <div
        ref={menuBgRef}
        className={`fixed inset-0 z-40 bg-neutral-900 flex items-center justify-center md:justify-start md:pl-32
        ${menuOpen ? "pointer-events-auto" : "pointer-events-none"}`}
      >
        <div className="flex flex-col space-y-4 md:space-y-8">
          {menuLinks.map((item, index) => (
            <div key={item.label} ref={(el) => (menuItemsRef.current[index] = el)} className="group relative block w-fit">

                className="inline-block text-4xl md:text-7xl font-black text-white/40 transition-all duration-500 hover:text-[#f8d249] hover:translate-x-6">

      </div>

      <main className="pt-32 px-6 md:px-12 max-w-7xl mx-auto space-y-20 pb-40">

          <h1 className="text-6xl md:text-9xl font-black leading-tight">
            REVERSE <br />
